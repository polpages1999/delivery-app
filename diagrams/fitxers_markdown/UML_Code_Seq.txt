@startuml
' ***** Declaració d'actors i objectes. 
' ***** Els objectes poden ser comunicació (boundary), control o entity
actor Desenvolupador
boundary CrearPlataforma
control Cont_Plat
entity Plataforma

Desenvolupador -> CrearPlataforma : CreacioPlat()


CrearPlataforma -> Cont_Plat : DesenvPlat()


actor Botiguer
entity Presentacio_oferta
actor Client
actor Transportista
actor Revisor_reembolso
Entity Selecciona_comanda
entity Solicitud_comanda
boundary Saber_Estat
control Estat
Entity Estat_comanda
Entity Estat_comanda
Entity Recepcio_comanda
boundary Rev
control Estat_Rev
Entity Peticio_revisio
entity Valoracio
actor Manteniment
Entity manteniment
Plataforma -> Selecciona_comanda: Perm_com()
Presentacio_oferta -> Selecciona_comanda: Anunci_comanda()
Botiguer -> Estat_comanda: Veure_Estat()
Client -> Selecciona_comanda: Sel_Com()
Cont_Plat -> Plataforma : Deixa_Com()
Selecciona_comanda -> Solicitud_comanda : Prep_com(s)
Solicitud_comanda -> Saber_Estat: Prep_Estat()
Saber_Estat -> Estat: Cont_Est()
Estat -> Estat_comanda: Visualitzar_Estat()
Transportista -> Estat_comanda: Transport_Estat()
Estat_comanda -> Recepcio_comanda: Rep_comanda()
Revisor_reembolso -> Rev: Mirar_revision()
Rev -> Estat_Rev: Arreglar_producte()
Estat_Rev -> Peticio_revisio: Enviar_prod_rev()
alt Estat_wrong
    Recepcio_comanda -> Peticio_revisio: Demanar_Rev()
end
alt Valorar
    Recepcio_comanda -> Valoracio: Valorar_servei()
end
Manteniment -> manteniment: solucionar_plat()
loop errors_greus
manteniment -> Cont_Plat : arreglar_errors()
end
@enduml
